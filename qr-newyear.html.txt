<script>
setTimeout(()=>{
 document.getElementById("qr").style.display="none";
 document.getElementById("main").style.display="block";
},4000);

const c=document.getElementById("c");
const x=c.getContext("2d");
c.width=innerWidth;
c.height=innerHeight;

let hearts=[];
function H(){
 this.x=Math.random()*c.width;
 this.y=c.height;
 this.s=Math.random()*20+10;
 this.v=Math.random()*2+1;
}
function draw(h){
 x.beginPath();
 x.moveTo(h.x,h.y);
 x.bezierCurveTo(h.x-h.s,h.y-h.s,h.x-h.s*2,h.y+h.s,h.x,h.y+h.s*2);
 x.bezierCurveTo(h.x+h.s*2,h.y+h.s,h.x+h.s,h.y-h.s,h.x,h.y);
 x.fillStyle="pink";
 x.fill();
}
function anim(){
 x.clearRect(0,0,c.width,c.height);
 hearts.forEach((h,i)=>{
  draw(h); h.y-=h.v;
  if(h.y<0) hearts.splice(i,1);
 });
 requestAnimationFrame(anim);
}
setInterval(()=>hearts.push(new H()),200);
anim();
</script>
